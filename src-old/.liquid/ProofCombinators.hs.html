 64: -- | proof casting
 65: -- | `x *** QED`: x is a proof certificate* strong enough for SMT to prove your theorem
 66: -- | `x *** Admit`: x is an unfinished proof

 68: infixl 3 ***

 79: -------------------------------------------------------------------------------
 80: -- | * Checked Proof Certificates ---------------------------------------------
 81: -------------------------------------------------------------------------------
 82: 
 83: -- Any (refined) carries proof certificates.
 84: -- For example 42 :: {v:Int | v == 42} is a certificate that
 85: -- the value 42 is equal to 42.
 86: -- But, this certificate will not really be used to proof any fancy theorems.
 87: 
 88: -- Below we provide a number of equational operations
 89: -- that constuct proof certificates.
 90: 
 91: -- | Implicit equality
 92: 
 93: -- x === y returns the proof certificate that
 94: -- result value is equal to both x and y
 95: -- when y == x (as assumed by the operator's precondition)
 96: 
 97: infixl 3 ===
 98: {-@ (===) :: x:a -> y:{a | y == x} -> {v:a | v == x && v == y} @-}
 99: (===) :: a -> a -> a
100: {VV : a | VV == y}_ forall a .
x1:a -> x2:{VV : a | VV == x1} -> {VV : a | VV == x1
                                            && VV == x2}=== ay  = y
101: 
102: infixl 3 =<=
103: {-@ (=<=) :: x:a -> y:{a | x <= y} -> {v:a | v == y} @-}
104: (=<=) :: a -> a -> a
105: {VV : a | VV == y}_ forall a . x1:a -> x2:{VV : a | x1 <= VV} -> {VV : a | VV == x2}=<= ay  = y
106: 
107: infixl 3 =>=
108: {-@ (=>=) :: x:a -> y:{a | x >= y}  -> {v:a | v == y} @-}
109: (=>=) :: a -> a -> a
110: {VV : a | VV == y}_ forall a . x1:a -> x2:{VV : a | x1 >= VV} -> {VV : a | VV == x2}=>= ay  = y
